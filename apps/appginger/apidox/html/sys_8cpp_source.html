00051 Ref * sysFastGetFastIterator( Ref * pc, class MachineClass * vm ) {
00052     Ref r = vm->fastPop();
00053     vm->checkStackRoom( 3 );
00054     if ( IsObj( r ) ) {
00055         Ref * obj_K = RefToPtr4( r );
00056         Ref key = *obj_K;
00057         if ( IsSimpleKey( key ) ) {
00058             switch ( KindOfSimpleKey( key ) ) {
00059                 case PAIR_KIND: {
00060                     vm->fastPush( r );          //  Iteration state.
00061                     vm->fastPush( sys_absent ); //  Iteration context, a dummy.
00062                     vm->fastPush( vm->sysFastListIterator() );
00063                     break;
00064                 }
00065                 case RECORD_KIND: {
00066                     throw ToBeDone();
00067                     break;
00068                 }
00069                 case STRING_KIND: {
00070                     vm->fastPush( LongToRef(0) );   //  Iteration state.
00071                     vm->fastPush( r );              //  Iteration context, a dummy.
00072                     vm->fastPush( vm->sysFastStringIterator() );
00073                     break;
00074                 }
00075                 case MAP_KIND: {
00076                 
00077                     //  Explode the map.
00078                     Ref * mark = vm->vp;
00079                     vm->fastPush( r );
00080                     vm->count = 1;
00081                     pc = sysMapExplode( pc, vm );
00082                     ptrdiff_t n = vm->vp - mark;
00083 
00084                     //  Gather the map as a vector.
00085                     vm->count = n;
00086                     pc = sysNewVector( pc, vm );
00087                     Ref x = vm->fastPop();
00088                     
00089                     //  Now treat it as a vector iteration.
00090 
00091                     vm->fastPush( LongToRef(1) );   //  Iteration state.
00092                     vm->fastPush( x );              //  Iteration context, a dummy.
00093                     vm->fastPush( vm->sysFastVectorIterator() );
00094                     break;
00095                 }
00096                 case VECTOR_KIND: {
00097                     vm->fastPush( LongToRef(1) );   //  Iteration state.
00098                     vm->fastPush( r );              //  Iteration context, a dummy.
00099                     vm->fastPush( vm->sysFastVectorIterator() );
00100                     break;
00101                 }
00102                 default: {
00103                     throw ToBeDone();
00104                 }
00105             }
00106         } else if ( IsObj( key ) ) {
00107             //  Compound keys not implemented yet.
00108             throw ToBeDone();   //  (compound keys)
00109         } else {
00110             throw ToBeDone();
00111         }
00112     } else if ( IsNil( r ) ) {
00113         vm->fastPush( r );          //  Iteration state.
00114         vm->fastPush( sys_absent ); //  Iteration context, a dummy.
00115         vm->fastPush( vm->sysFastListIterator() );
00116     } else {
00117         throw ToBeDone();
00118     }
00119     return pc;
00120 }

00157     if ( vm->count == 1 ) {
00158         Ref r = vm->fastPeek();
00159         if ( IsObj( r ) ) {
00160             Ref * obj_K = RefToPtr4( r );
00161             Ref key = *obj_K;
00162             if ( IsFunctionKey( key ) ) {
00163                 throw Mishap( "Trying to explode (...) a function object" );
00164             } else if ( IsSimpleKey( key ) ) {
00165                 switch ( KindOfSimpleKey( key ) ) {
00166                     case VECTOR_KIND: {
00167                         pc = sysVectorExplode( pc, vm );
00168                         break;
00169                     }
00170                     case PAIR_KIND: {
00171                         pc = sysListExplode( pc, vm );
00172                         break;
00173                     }
00174                     case RECORD_KIND: {
00175                         vm->fastDrop( 1 );
00176                         unsigned long n = sizeAfterKeyOfRecord( obj_K );
00177                         vm->checkStackRoom( n );
00178                         memcpy( vm->vp + 1, obj_K + 1, n * sizeof( Ref ) );
00179                         vm->vp += n;
00180                         break;
00181                     }
00182                     case STRING_KIND: {
00183                         pc = sysStringExplode( pc, vm );
00184                         break;
00185                     }
00186                     case MAP_KIND: {
00187                         pc = sysMapExplode( pc, vm );
00188                         break;
00189                     }
00190                     default: {
00191                         throw ToBeDone();
00192                     }
00193                 }
00194             } else if ( IsObj( key ) ) {
00195                 //  Compound keys not implemented yet.
00196                 throw ToBeDone();   //  (compound keys)
00197             } else {
00198                 throw ToBeDone();
00199             }
00200         } else if ( IsNil( r ) ) {
00201             return sysListExplode( pc, vm );
00202         } else {
