00080 void refPrint( std::ostream & out, Ref r ) {
00081     if ( IsObj( r ) ) {
00082         Ref * obj_K = RefToPtr4( r );
00083         Ref key = *obj_K;
00084         if ( IsFunctionKey( key ) ) {
00085             out << "<function " << numInputsOfFn( obj_K ) << ":" << numOutputsOfFn( obj_K ) << ">";
00086         } else if ( IsSimpleKey( key ) ) {
00087             switch ( KindOfSimpleKey( key ) ) {
00088                 case VECTOR_KIND: {
00089                     refVectorPrint( out, obj_K );
00090                     break;
00091                 }
00092                 case PAIR_KIND: {
00093                     refListPrint( out, r );
00094                     break;
00095                 }
00096                 case MAP_KIND: {
00097                     refMapPrint( out, obj_K );
00098                     break;
00099                 }
00100                 case RECORD_KIND: {
00101                     refRecordPrint( out, obj_K );
00102                     break;
00103                 }
00104                 case STRING_KIND: {
00105                     refStringPrint( out, obj_K );
00106                     break;
00107                 }               
00108                 default: {
00109                     throw "unimplemented (other)";
00110                 }
00111             }
00112         } else if ( IsObj( key ) ) {
00113             //  Compound keys not implemented yet.
00114             throw "unimplemented (compound keys)";
00115         } else {
00116             throw "unimplemented";
00117         }
00118     } else {
00119         Ref k = refKey( r );
00120         if ( k == sysSmallKey ) {
00121             out << SmallToLong( r );
00122         } else if ( k == sysBoolKey ) {
00123             out << ( r == sys_false ? "false" : "true" );
00124         } else if ( k == sysAbsentKey ) {
00125             out << "absent";
00126         } else if ( k == sysCharKey ) {
00127             out << CharacterToChar( r );
00128         } else if ( k == sysSymbolKey ) {
00129             out << symbolToStdString( r );
00130         } else if ( k == sysNilKey ) {
00131             refListPrint( out, r );
00132         } else if ( k == sysVectorKey ) {
00133             bool sep = false;
00134             out << "{";
00135             Ref * p = RefToPtr4( r );
00136             long len = RefToLong( p[ -1 ] );
00137             for ( int i = 1; i <= len; i++ ) {
00138                 if ( sep ) { out << ","; } else { sep = true; }
00139                 refPrint( p[ i ] ); 
00140             }
00141             out << "}";
00142         } else {
00143             out << "?(" << std::hex << ToULong( r ) << std::dec << ")";
00144         }
00145     }
