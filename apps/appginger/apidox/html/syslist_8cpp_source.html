00100     {
00101         //  May need to GC so leave on the stack.
00102         Ref rhs = vm->fastPeek();
00103         Ref lhs = vm->fastPeek( 1 );
00104         
00105         //  Typecheck arguments.
00106         if ( !isList( lhs ) || !isList( rhs ) ) throw Mishap( "Invalid arguments in listAppend" );
00107         
00108         ptrdiff_t start = vm->stackLength();

00110             vm->checkStackRoom( 1 );
00111             vm->fastPush( fastPairHead( lhs ) );
00112             lhs = fastPairTail( lhs );
00113         }
00114         vm->checkStackRoom( 1 );
00115         vm->fastPush( rhs );

00123     Ref r = vm->fastPop();
