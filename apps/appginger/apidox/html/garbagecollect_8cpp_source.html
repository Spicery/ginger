00392     void forward( Ref & current ) {
00393         gclogger.atRef( current );
00394         if ( not IsObj( current ) ) {
00395             if ( IsSymbol( current ) ) {
00396                 gcTouchSymbol( current );
00397             }
00398         } else {
00399             Ref * obj = RefToPtr4( current );
00400             if ( IsCoreFunctionKey( *obj ) ) return;
00401             if ( IsFwd( *obj ) ) {
00402                 //  Already forwarded.
00403                 //  Pick up the to-space address.
00404                 current = Ptr4ToRef( FwdToPtr4( *obj ) );
00405             } else {
00406                 Ref * half_copied_obj  = copier.copy( obj );
00407                 current = Ptr4ToRef( half_copied_obj );
00408                 
00409                 //  We stomp on the key with a forwarded value.
00410                 *obj = Ptr4ToFwd( half_copied_obj );
00411             }
00412         }
00413     }

00429     
00430         //  The PC might be set to a fake value.
00431         if ( this->vm->func_of_program_counter != NULL ) {
00432             this->vm->func_of_program_counter = this->forwardPtr4( this->vm->func_of_program_counter );
00433         }
00434         
00435         CallStackCrawl csc( this->vm );

00437             Ref * sp = csc.next();

00439             if ( sp[ SP_FUNC ] == 0 ) break;

00443             
00444             Ref * new_func = forwardPtr4( func );
00445             sp[ SP_FUNC ] = new_func;
00446             sp[ SP_LINK ] = new_func + d;
00447             
00448             unsigned long n = NSLOTS( sp );
00449             for ( unsigned long i = 0; i < n; i++ ) {
00450                 this->forward( LOCAL_OF( sp, i ) );
00451             }           
00452         }

00488         if ( IsFunctionKey( key ) ) {
00489             if ( IsHeapFunctionKey( key ) ) {
