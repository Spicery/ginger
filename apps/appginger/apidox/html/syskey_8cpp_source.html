00024 Ref refKey( Ref r ) {
00025     unsigned long u = ( unsigned long )r;
00026     unsigned long tag, tagg, taggg;
00027     tag = u & TAG_MASK;
00028     if ( tag == INT_TAG ) return sysSmallKey;
00029     if ( tag == OBJ_TAG ) return *RefToPtr4(r);
00030     tagg = u & TAGG_MASK;
00031     if ( tagg == ( 0 | SIM_TAG ) ) return sysAbsentKey;
00032     if ( tagg == ToULong( sys_false ) || tagg == ToULong( sys_true ) ) return sysBoolKey;
00033     if ( tagg == FN_TAGG ) return sysFunctionKey;
00034     if ( tagg == KEY_TAGG ) return sysKeyKey;
00035     taggg = u & TAGGG_MASK;
00036     if ( taggg == CHAR_TAGGG ) return sysCharKey;
00037     if ( taggg == SYMBOL_TAGGG ) return sysSymbolKey;
00038     if ( taggg == MISC_TAGGG ) {
00039         if ( r == sys_nil ) {
00040             return sysNilKey; 
00041         } else {
00042             throw;
00043         }
00044     }
00045     throw;
00046 }
