00052 static long log2( unsigned long v ) {
00053     unsigned long t, tt, w;     // temporaries
00054     
00055     if (( w = v >> 32 )) {
00056         if (( tt = w >> 16 )) {
00057             return ( t = tt >> 8 ) ? 56 + LogTable256[ t ] : 48 + LogTable256[ tt ];
00058         } else {
00059             return ( t = w >> 8 ) ? 40 + LogTable256[ t ] : 32 + LogTable256[ w ];
00060         }
00061     } else if (( tt = v >> 16 )) {
00062         return ( t = tt >> 8 ) ? 24 + LogTable256[ t ] : 16 + LogTable256[ tt ];
00063     } else {
00064         return ( t = v >> 8 ) ? 8 + LogTable256[ t ] : LogTable256[ v ];
00065     }
00066 }

00177 /*
00178     - Takes N arguments.
00179     - Estimates size for an initial map by making a preliminary pass 
00180     over the N arguments.
00181     - Builds the initial map.
00182     - Then adds the entries in a second pass over the N arguments.
00183 */
00184 Ref * sysNewMap( Ref *pc, MachineClass * vm ) {
00185     long count = estimateMapSize( vm );
00186     //std::cout << "Estimated size is " << count << std::endl;
00187     
00188     //  Calculate the total preflight size: we have a map record (3 long words),
00189     //  a data vector (2 + power of 2 greater than count), and count bucket 
00190     //  records (each 4 long words).
00191     long width = 1 + log2( count );
00192     long data_size = 1 << width;
00193     long preflight = 3 + ( 2 + data_size ) + ( 4 * count );
00194     //std::cout << "Estimated preflight size as " << preflight << std::endl;
00195 
00196     XfrClass xfr( vm->heap().preflight( pc, preflight ) );
00197     
00198     xfr.xfrRef( LongToSmall( data_size ) );
00199     xfr.setOrigin();
00200     xfr.xfrRef( sysVectorKey );
00201     
00202     //  ToBeDone: We could really do with a memcpy style operation here. 
00203     for ( long i = 0; i < data_size; i++ ) {
00204         xfr.xfrRef( sys_absent );
00205     }
00206     
00207     Ref * data_refref = xfr.makeRefRef();
00208     
00209     xfr.setOrigin();
00210     xfr.xfrRef( sysMapKey );    
00211     xfr.xfrRef( Ptr4ToRef( data_refref ) );
00212     xfr.xfrRef( LongToSmall( count ) );
00213     Ref map_ref = xfr.makeRef();
00214     
00215     //  Now add all the members. Need to perform the update in LEFT to RIGHT
00216     //  order, from vm->vp[ -( vm->count - 1 ) ] to vm->vp[ 0 ]
00217     AddArgument args( xfr, data_refref + 1, data_size, width );
00218     for ( long i = vm->count - 1; i >= 0; i-- ) {
00219         Ref r = *( vm->vp - i );
00220         args.add( r );
00221     }
00222     
00223     RefToPtr4( map_ref )[ MAP_COUNT_OFFSET ] = LongToSmall( args.numBuckets() );
00224     
00225     vm->vp -= vm->count;
00226     *( ++vm->vp ) = map_ref;
00227     
00228     return pc;
00229 }
