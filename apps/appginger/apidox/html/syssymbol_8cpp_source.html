00051 static unsigned long nextSerial() {
00052     bool resetted = false;
00053     for (;;) {
00054         if ( available( next_serial ) ) return next_serial++ ;
00055         if ( next_serial > MAX_SERIAL ) {
00056             if ( resetted ) {
00057                 throw Mishap( "Symbol table is full" );
00058             } else { 
00059                 next_serial = 0;
00060                 resetted = true;
00061             }
00062         }
00063     }
00064 }

00095 Ref refMakeSymbol( const std::string & s ) {
00096     unsigned serial;
00097     SYMTAB::const_iterator it = symtab.find( s );
00098     if ( it == symtab.end() ) {
00099         serial = nextSerial();
00100         symtab.insert( std::pair< const std::string, unsigned int >( s, serial ) );
00101         revsymtab.insert( std::pair< unsigned int, const std::string >( serial, s ) );
00102     } else {
00103         serial = it->second;
00104     }
00105     return ToRef( ( serial << TAGGG ) | SYMBOL_TAGGG );
00106 }
