00068 Ref HeapClass::copyString( const char *s ) {
00069     //  Strings have a two Ref overhead & must be ref aligned and have
00070     //  an additional 1-byte overhead for null-byte.
00071     size_t n = strlen( s );
00072     Ref * fake_pc = static_cast< Ref * >( 0 );
00073     XfrClass xfr( fake_pc, *this, 2 + ( n + 1 + sizeof( Ref ) - 1 ) / sizeof( Ref ) );
00074     
00075     //  We will actually copy n+1 bytes to include the null-termination.
00076     //  So the calculation of the size is a little bit trickier.
00077     xfr.xfrRef( LongToRef( n ) );
00078     xfr.setOrigin();
00079     xfr.xfrRef( sysStringKey );

00081     xfr.xfrSubstring( s, 0, n );
