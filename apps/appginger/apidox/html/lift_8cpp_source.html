00059 class Env {
00060 private:
00061     std::vector< Scope > idents;
00062 
00063 public: 
00064     void add( Ident id ) {
00065         this->idents.back().data.push_back( id );
00066     }
00067     
00068     FnTermClass * getFunction( const int n ) const {
00069         return this->idents[ n ].fn;
00070     }
00071     
00072     void setAsFinalInput( int from_index, Ident new_input ) {
00073         FnTermClass * f = this->idents[ from_index ].fn;
00074         int & nin = f->ninputs();
00075     
00076         nin += 1;
00077         int slot = new_input->getFinalSlot();
00078         cerr << "Bumping #inputs for " << f->name() << " from " << nin-1 << " to " << nin << endl;
00079         
00080         //  Now we need to swap nin and slot. Check there is a need to make
00081         //  changes!
00082         if ( nin != slot ) {    
00083             int N = this->idents.size();
00084             for ( int i = from_index; i < N; i++ ) {
00085                 Scope & le = this->idents[ i ];
00086                 if ( le.fn != NULL && i > from_index ) break;
00087                 
00088                 std::vector< Ident > & v = le.data;
00089                 for ( std::vector< Ident >::iterator it = v.begin(); it != v.end(); ++it ) {
00090                     it->get()->swapSlot( nin, slot );
00091                 }
00092             }
00093         }
00094     }
00095         
00096     void capturingFunctions( Ident & outer, std::list< int >  & index_of_functions ) {
00097         for ( int i = this->idents.size() - 1; i >= 0; i-- ) {
00098             Scope & le = this->idents[ i ];
00099             FnTermClass * f = le.fn;
00100             if ( f ) {
00101                 index_of_functions.push_front( i );
00102             }
00103             if ( f == outer->function() ) return;
00104         }       
00105         throw Unreachable();
00106     }
00107     
00108     Ident search( const std::string & c ) {
00109         for ( std::vector< Scope >::reverse_iterator it = this->idents.rbegin(); it != this->idents.rend(); ++it ) {
00110             Scope & le = *it;
00111             std::vector< Ident > & v = le.data;
00112             for ( std::vector< Ident >::reverse_iterator jt = v.rbegin(); jt != v.rend(); ++jt ) {
00113                 Ident ident = *jt;
00114                 const std::string & ic = ident->getNameString();
00115                 #ifdef DBG_LIFT
00116                     fprintf( stderr, "Comparing %s with %s\n", c.c_str(), ic.c_str() );
00117                 #endif
00118                 if ( c == ic ) {
00119                     return ident;
00120                 }       
00121             }
00122         }
00123         return shared< IdentClass >();
00124     }
00125     
00126     void mark( FnTermClass * fn ) {
00127         this->idents.push_back( Scope( fn ) );
00128     }
00129     
00130     void cutToMark() {
00131         this->idents.pop_back();
00132     }

00164 static void lookupAndAddGlobal( Package * current, VarTermClass * t ) {
00165     const std::string & pkg = t->pkg();
00166     const std::string & c = t->name();
00167     const FacetSet * facets = t->facets();
00168     switch ( t->refType() ) {
00169         case LOCAL_REF_TYPE: {
00170             Package * p = current;
00171             t->ident() = p->lookup_or_add( c, facets );
00172             break;
00173         }
00174         case ABSOLUTE_REF_TYPE: {
00175             Package * p = current->getPackage( pkg );
00176             t->ident() = p->lookup_or_add( c, facets );
00177             break;
00178         }
00179         case ALIAS_REF_TYPE: {
00180             Import * imp = current->getAlias( t->alias() );
00181             const FacetSet * m = imp->matchingTags();
00182         
00183             //cout << "ALIAS DECLARATION" << endl;
00184             //cout << "Import facet : " << *m << endl;
00185             //cout << "Declaration facets: " << *facets << endl;
00186         
00187             if ( imp != NULL && facets->isntEmptyIntersection( m ) ) {
00188                 Ident id = imp->package()->lookup_or_add( c, facets );
00189                 t->ident() = id;
00190             } else if ( imp != NULL ) {
00191                 throw Mishap( "Declaration in package referenced by alias of an import would not be exported" );
00192             } else {
00193                 throw Mishap( "No such alias" );
00194             }
00195             break;
00196         }
00197     }

00200 static Ident lookupGlobal( Package * current, IdTermClass * t ) {
00201     const std::string & pkg = t->pkg();
00202     const std::string & c = t->name();
00203     switch ( t->refType() ) {
00204         case LOCAL_REF_TYPE: {
00205             return current->lookup( c, true );
00206         }
00207         case ABSOLUTE_REF_TYPE: {
00208             Package * p = current->getPackage( pkg );
00209             return p->lookup( c, true );
00210         }
00211         case ALIAS_REF_TYPE: {
00212             Import * imp = current->getAlias( t->alias() );
00213             if ( imp != NULL ) {
00214                 Ident id = imp->package()->lookup( c, true );
00215                 const FacetSet * m = imp->matchingTags();
00216                 if ( id->facets->isntEmptyIntersection( m ) ) {
00217                     return id;
00218                 } else {
00219                     throw Mishap( "Aliased variable not exported" );
00220                 }
00221             } else {
00222                 throw Mishap( "No such alias" );
00223             }
00224             throw Unreachable();
00225         }
00226         default:
00227             throw Unreachable();
00228     }

00232 enum Lookup LiftStateClass::lookup( IdTermClass * t, Ident *id ) {
00233     const std::string & c = t->name();
00234     Ident ident = this->env.search( c ); 
00235     if ( ident ) {
00236         *id = ident;
00237         #ifdef DBG_LIFT
00238             fprintf( stderr, "< lookup\n" );
00239         #endif
00240         Lookup result = ident->level >= this->level ? InnerLocal : OuterLocal;
00241         if ( result == OuterLocal ) {
00242             ident->setOuter();
00243         }
00244         return result;
00245     } else {
00246         *id = lookupGlobal( this->package, t );
00247         if ( *id == 0 ) throw Mishap( "Undeclared variable" ).culprit( "Variable", c );
00248         #ifdef DBG_LIFT
00249             fprintf( stderr, "< lookup\n" );
00250         #endif
00251         return Global;
00252     }

00321             //  A single variable var(x) or

00325                 case fnc_var: {
00326                     fn->ninputs() = 1;
00327                     NamedTermMixin * arg = dynamic_cast< NamedTermMixin * >( args.get() );
00328                     this->addIdent( arg, fn, slot );
