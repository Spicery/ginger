00298 static void tprint( Term term, int level ) {
00299     splevel( level );
00300     printf( "[%d,%s]", term->count(), term->type_name() );  //  DEBUG ONLY
00301     printf( "_%s_\n", functor_name( term->functor() ) );
00302 
00303     switch ( term->functor() ) {
00304         case fnc_int: {
00305             splevel( level + 1 );
00306             printf( "%d\n", term_int_cont( term ) );
00307             break;
00308         }
00309         case fnc_string: {
00310             splevel( level + 1 );
00311             printf( "%s\n", term_string_cont( term ) );
00312             break;
00313         }
00314         case fnc_id: {
00315             const std::string & c = term->name();
00316             Ident & id = term->ident();
00317             splevel( level + 1 );
00318             printf(
00319                 "%s, length = %Zu, ",
00320                 c.c_str(), c.size()
00321             );
00322             if ( id == NULL ) {
00323                 printf( "slot = <null>" );
00324             } else if ( id->isLocal() ) {
00325                 printf( "slot = %d", id->getFinalSlot() );
00326             } else {
00327                 printf( "global" );
00328             }
00329             printf( "\n" );
00330             break;
00331         }
00332         default: {
00333             //  printf( "(%d children to print)\n", term->count );
00334             {
00335                 const int n = term_count( term );
00336                 for ( int i = 0; i < n; i++ ) {
00337                     tprint( term_index( term, i ), level + 1 );
00338                 }
00339             }
00340         }
00341     }
