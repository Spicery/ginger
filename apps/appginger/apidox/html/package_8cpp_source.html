00183 void Package::import( const Import & imp ) {
00184     //  Strictly speaking we should not pushback until we have verified
00185     //  we have a non-duplicate import. Imports are indexed by
00186     //      -   facet
00187     //      -   from package
00188     //      -   alias
00189     for ( 
00190         std::vector< Import >::iterator it = this->imports.begin(); 
00191         it != this->imports.end();
00192         ++it
00193     ) {
00194         if ( 
00195             it->matching_tags == imp.matching_tags &&
00196             it->from == imp.from &&
00197             it->alias == imp.alias
00198         ) {
00199             *it = imp;
00200             return;
00201         }
00202     }
00203     //  No previous matching imports, so add.
00204     this->imports.push_back( imp );
