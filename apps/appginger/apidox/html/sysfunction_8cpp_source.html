00027 /*
00028     This function is called with the stack looking like
00029     this:
00030         item1, ..., itemN, F
00031     It delivers a function which is equivalent to Pop-11's
00032         F(% item1, ..., itemN %)
00033         
00034         
00035     Note that the final push is entirely safe because there must have
00036     been some N+1 arguments.
00037 */
00038 Ref * sysPartApply( Ref *pc, class MachineClass * vm ) {
00039     PlantClass * plant = vm->plant();
00040     
00041     //  We have to do a heap check immediately. This is intended to 
00042     //  ensure that garbage collection cannot occur before we exit this
00043     //  function & we can therefore not worry about creating temporary roots.
00044     //
00045     //  We exploit the fact that we can count the size of the PUSHQ and
00046     //  CHAIN_LITE instructions.
00047     XfrClass xfr( pc, vm->heap(), static_cast< int >( OFFSET_FROM_FN_LENGTH_TO_KEY + 1 + 2 * vm->count + 1 ) );
00048     
00049     Ref F = vm->fastPeek();
00050     
00051     //  F is required to be a function.
00052     if ( !IsObj( F ) ) throw Mishap( "Function needed as last arg to partApply" );
00053     Ref * F_K = RefToPtr4( F );
00054     if ( !IsFunctionKey( *F_K ) ) throw Mishap( "Function needed as last arg to partApply" );
00055     
00056     long F_in_arity = numInputsOfFn( F_K );
00057     long F_out_arity = numOutputsOfFn( F_K );
00058     
00059     
00060     int N = vm->count - 1;
00061     vm->vp -= N;                //  Drop the N items, items to be accessed by vm->vp[i]
00062 
00063     //  This seems very unlikely to be correct. We can't just go around
00064     //  substracting arities.
00065     vmiFUNCTION( plant, F_in_arity - N, F_out_arity );
00066     for ( int i = 0; i < N; i++ ) {
00067         vmiPUSHQ( plant, vm->vp[ i ] );
00068     }
00069     vmiCHAIN_LITE( plant, F, N );
00070     Ref part_apply = vmiENDFUNCTION( plant );
00071     
00072     vm->fastPeek() = part_apply;
00073 
00074     
00075     return pc;
00076 }
